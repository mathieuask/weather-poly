# Weather-poly automated tasks
# Install:  crontab /Users/mathieuaskamp/Desktop/weather-poly/weather-poly/backend/crontab.txt
# Verify:   crontab -l
# Disable:  crontab -r

REPO=/Users/mathieuaskamp/Desktop/weather-poly/weather-poly
VENV=/Users/mathieuaskamp/Desktop/weather-poly/weather-poly/backend/.venv/bin/python3

# Scanner — toutes les 30 min
*/30 * * * * cd $REPO && $VENV backend/scanner.py >> scanner.log 2>&1

# Tracker — toutes les 30 min, 2 min après scanner
2,32 * * * * cd $REPO && $VENV backend/tracker.py >> tracker.log 2>&1

# METAR intraday — toutes les 15 min (6h-22h UTC)
*/15 6-22 * * * cd $REPO && $VENV backend/metar.py >> metar.log 2>&1

# Calibration — toutes les 6h
0 0,6,12,18 * * * cd $REPO && $VENV backend/calibrate.py >> calibrate.log 2>&1

# Health check — toutes les heures
0 * * * * cd $REPO && $VENV backend/healthcheck.py >> healthcheck.log 2>&1

# Rapport quotidien Telegram — 8h UTC
0 8 * * * cd $REPO && $VENV backend/autoresearch.py >> autoresearch.log 2>&1

# Archive ensemble — 4x/jour après les runs GFS
30 4,10,16,22 * * * cd $REPO && $VENV backend/archive_ensemble.py >> archive.log 2>&1

# Collecte backtest — 1x/jour à 3h UTC
0 3 * * * cd $REPO && $VENV backend/collect.py --days 3 >> collect.log 2>&1

# Export données frontend — 1x/jour à 4h UTC
0 4 * * * cd $REPO && $VENV backend/export_data.py >> export.log 2>&1
